\chapter{Cartesian Products and Relations}

\text{For two sets \A and \B, the Cartesian Product of \A and \B is\\}
$$A \times B = \{(a, b); a \in A, b \in B\}$$
\text{We say that the elements of \A \(\times\) \B are ordered pairs\\}

\begin{definition}
: For sets A, B, any subset of \A \(\times\) \B is called a binary relation from
\A to \B. Any subset of \A \(\times\) \A is called a \textbf{binary relation} on \A.
\end{definition}
